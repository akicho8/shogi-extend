def run
  @rows.clear
  perform
  if params[:developper_notice]
    ApplicationMailer.developper_notice(subject: subject, body: mail_body).deliver_now
  end
  if Rails.env.development?
    puts params.to_t
    puts @rows.to_t
  end
  self
end
