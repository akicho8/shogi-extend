def icon_params(grade_diff)
  if icon_key
    { :key => icon_key, :class => icon_class }
  else
    case
    when grade_diff >= 1
      { :key => "emoticon-neutral-outline", :class => icon_class }
    when grade_diff == 0
      { :key => "emoticon-sad-outline", :class => icon_class }
    else
      { :key => "emoticon-dead-outline", :class => icon_class }
    end
  end
end
