def icon_params(grade_diff)
  if icon_key
    { key: icon_key, :class => icon_class }
  else
    if key == :win
      case
      when grade_diff >= 1
        { key: "star-circle", :class => "has-text-gold", }
      when grade_diff == 0
        { key: "star", :class => "has-text-gold", }
      else
        { key: "star-outline", :class => "has-text-gold", }
      end
    end
  end
end
